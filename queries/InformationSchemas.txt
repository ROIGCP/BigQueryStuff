CREATE TABLE datasetdemo.names 
  as SELECT * FROM roigcp.demo.names

  date(tscolumn)
  as SELECT * FROM roigcp.demo.names

  CLUSTER BY lastname,firstname
  as SELECT * from roigcp.demo.names

CREATE TABLE datasetdemo.namespartitionandcluster
  PARTITION BY date(tscolumn)
  as SELECT * FROM roigcp.demo.names
